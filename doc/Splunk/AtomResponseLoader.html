153:       value = {}
154:       for child in node.children
155:         if child.node_type_name.eql?('text')
156:           text = child.content
157:           return nil if text.nil?
158:           text.strip!
159:           next if text.size == 0
160:           next if node.children.size > 1
161:           return text
162:         elsif child.node_type_name.eql?('element')

164:           return load_list(child) if is_list(child)
